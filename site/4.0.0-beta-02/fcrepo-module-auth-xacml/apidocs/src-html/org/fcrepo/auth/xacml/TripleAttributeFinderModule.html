055/**
056 * Finds resource attributes based on triples in the Fedora graph. Retrieves values where the attribute URI matches the
057 * triple predicate and the triple object can be supplied as the requested data type.
058 *
059 * @author Gregory Jansen
060 * @author Andrew Woods
061 * @author Scott Prater
062 */
063@Component("tripleAttributeFinderModule")
064public class TripleAttributeFinderModule extends AttributeFinderModule {
065
066    private static final Logger LOGGER = getLogger(TripleAttributeFinderModule.class);
067
068    private static BagAttribute empty_bag;
069
070    private static IdentifierTranslator idTranslator;
071
072    /**
073     * Fedora's ModeShape session factory.
074     */
075    @Autowired
076    protected SessionFactory sessionFactory;
077
078    @Autowired
079    protected NodeService nodeService;
080
081    /**
082     * Supported designator types.
083     */
084    private static final Set<Integer> DESIGNATOR_TYPES = unmodifiableSet(singleton(RESOURCE_TARGET));
085
086    /**
087     * Supports designators.
088     *
089     * @return if designator is supported.
090     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#
091     * isDesignatorSupported()
092     */
093    @Override
094    public final boolean isDesignatorSupported() {
095        return true;
096    }
097
098    /**
099     * Supports resource attributes.
100     *
101     * @return the supported designator types.
102     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#getSupportedDesignatorTypes()
103     */
104    @Override
105    public final Set<Integer> getSupportedDesignatorTypes() {
106        return DESIGNATOR_TYPES;
107    }
108
109    /**
110     * Finds the matching triples values.
111     *
112     * @attributeId The URI of the attribute key (the predicate of the triple)
113     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#findAttribute (java.net.URI, java.net.URI,
114     *      java.net.URI, java.net.URI, org.jboss.security.xacml.sunxacml.EvaluationCtx, int)
115     */
116    @Override
117    public final EvaluationResult findAttribute(final URI attributeType,
118                                                final URI attributeId,
119                                                final URI issuer,
120                                                final URI subjectCategory,
121                                                final EvaluationCtx context,
122                                                final int designatorType) {
123        LOGGER.debug("findAttribute({}, {}, {}, {}, {}, {})",
124                     attributeType, attributeId, issuer, subjectCategory, context, designatorType);
125
126        empty_bag = createEmptyBag(attributeType);
127
128        // Make sure this is a Resource attribute
129        if (designatorType != RESOURCE_TARGET) {
130            LOGGER.debug("Not looking for a resource attribute");
131            return new EvaluationResult(empty_bag);
132        }
133
134        final Session session;
135        try {
136            session = sessionFactory.getInternalSession();
137        } catch (final RepositoryException e) {
138            LOGGER.debug("Error getting session!");
139            final Status status = new Status(singletonList(STATUS_PROCESSING_ERROR), "Error getting session");
140            return new EvaluationResult(status);
141        }
142
143        // The resourceId is the path of the object be acted on, retrieved from the PDP evaluation context
144        final EvaluationResult ridEvalRes =
145                context.getResourceAttribute(URI.create("http://www.w3.org/2001/XMLSchema#string"),
146                        URIConstants.ATTRIBUTEID_RESOURCE_ID, null);
147        final AttributeValue resourceIdAttValue = ridEvalRes.getAttributeValue();
148        if (resourceIdAttValue.getValue().toString().isEmpty()) {
149            LOGGER.debug("Context should have a resource-id attribute!");
150            final Status status = new Status(singletonList(STATUS_PROCESSING_ERROR), "Resource Id not found!");
151            return new EvaluationResult(status);
152        }
153
154        String resourceId = (String) resourceIdAttValue.getValue();
155
156        // if dealing with set_property action, use parent node for triples
157        final Set<String> actions = PolicyUtil.getActions(context);
158        if (actions.contains("set_property") || actions.contains("add_node")) {
159            resourceId = resourceId.substring(0, resourceId.lastIndexOf("/{"));
160            if (resourceId.length() == 0) {
161                resourceId = "/";
162            }
163        }
164
165        // Get the resource to be acted on
166        final FedoraResource resource;
167        final String path;
168        try {
169            resource = nodeService.getObject(session, resourceId);
170            if (resource == null) {
171                LOGGER.debug("Cannot find a fedora resource for {}", resourceId);
172                return new EvaluationResult(empty_bag);
173            }
174            path = resource.getPath();
175            idTranslator = new DefaultIdentifierTranslator();
176
177        } catch (final RepositoryException e) {
178            // If the object does not exist, it may be due to the action being "create"
179            return new EvaluationResult(empty_bag);
180        }
181
182        LOGGER.debug("Looking for properties on modeshape path {} with repo path {}", resourceId, path);
183
184        // Get the properties of the resource
185        Model properties;
186        Resource graphNode;
187        try {
188            properties = resource.getTriples(idTranslator).asModel();
189            graphNode = idTranslator.getSubject(resource.getPath());
190        } catch (final RepositoryException e) {
191            LOGGER.debug("Cannot retrieve any properties for [{}]:  {}", resourceId, e);
192            final Status status =
193                    new Status(singletonList(STATUS_PROCESSING_ERROR),
194                            "Error retrieving properties for [" + path + "]!");
195            return new EvaluationResult(status);
196        }
197
198        LOGGER.debug("Looking for properties on graph node: {}", graphNode.getURI());
199
200        // Get the values of the properties matching the type
201        final Iterator<RDFNode> matches =
202                properties.listObjectsOfProperty(graphNode, properties.createProperty(attributeId.toString()));
203
204        final Set<AttributeValue> attr_bag = new HashSet<>();
205
206        // Add the properties to the bag
207        while (matches.hasNext()) {
208            final RDFNode match = matches.next();
209            final String uri = match.asResource().getURI();
210            LOGGER.debug("Found property: {}", uri);
211            attr_bag.add(new AnyURIAttribute(URI.create(uri)));
212        }
213
214        // Return the results, or any empty bag
215        if (attr_bag.isEmpty()) {
216            LOGGER.debug("No matching properties found");
217            return new EvaluationResult(empty_bag);
218        }
219
220        return new EvaluationResult(new BagAttribute(attributeType, attr_bag));
221    }
