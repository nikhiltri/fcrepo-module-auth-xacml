016package org.fcrepo.auth.xacml;
017
018import java.net.URI;
019
020import org.jboss.security.xacml.interfaces.XACMLConstants;
021
022
023/**
024 * URIs that are used in this module.
025 *
026 * @author Gregory Jansen
027 */
028public abstract class URIConstants {
029
030    /**
031     * ID of the subject (user principal).
032     */
033    public static final URI ATTRIBUTEID_SUBJECT_ID = URI
034            .create(XACMLConstants.ATTRIBUTEID_SUBJECT_ID);
035
036    public static final String FCREPO_SUBJECT_ROLE_VALUE = "fcrepo-xacml:subject-role";
037
038    public static final URI FCREPO_SUBJECT_ROLE = URI.create(FCREPO_SUBJECT_ROLE_VALUE);
039
040    /**
041     * ID of the action (ModeShape permission name).
042     */
043    public static final URI ATTRIBUTEID_ACTION_ID = URI
044            .create(XACMLConstants.ATTRIBUTEID_ACTION_ID);
045
046    /**
047     * ID of the resource (ModeShape node/property path).
048     */
049    public static final URI ATTRIBUTEID_RESOURCE_ID = URI
050            .create(XACMLConstants.ATTRIBUTEID_RESOURCE_ID);
051
052    /**
053     * External URI of the resource (External Identifier).
054     */
055    public static final URI ATTRIBUTEID_RESOURCE_URI = URI.create("urn:fedora:xacml:2.0:resource:resource-uri");
056
057    /**
058     * ID of the ModeShape workspace for this resource.
059     */
060    public static final URI ATTRIBUTEID_RESOURCE_WORKSPACE = URI
061            .create("urn:fedora:xacml:2.0:resource:resource-workspace");
062
063    /**
064     * Scope of the request (DESCENDANTS if "remove", IMMEDIATE otherwise).
065     */
066    public static final URI ATTRIBUTEID_RESOURCE_SCOPE = URI
067            .create("urn:oasis:names:tc:xacml:1.0:resource:scope");
068
069    public static final String XACML_POLICY_PROPERTY = "authz:policy";
070
071    public static final String POLICY_URI_PREFIX = "info:fedora";
072
073    /**
074     * Original request IP, which may be forwarded by an application.
075     */
076    public static final URI ATTRIBUTEID_ENVIRONMENT_ORIGINAL_IP_ADDRESS = URI
077            .create("urn:fedora:xacml:2.0:environment:original-ip-address");
078
079}
